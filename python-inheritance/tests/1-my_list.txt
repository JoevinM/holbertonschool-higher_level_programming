>>> MyList = __import__('1-my_list').MyList

# Empty list
>>> my_list = MyList()
>>> my_list.print_sorted()
[]

# Append elements and print
>>> my_list.append(10)
>>> my_list.append(5)
>>> my_list.append(7)
>>> print(my_list)
[10, 5, 7]

# Sorted view
>>> my_list.print_sorted()
[5, 7, 10]

# Original list remains unchanged
>>> print(my_list)
[10, 5, 7]

# Already sorted
>>> my_list = MyList([1, 2, 3, 4])
>>> my_list.print_sorted()
[1, 2, 3, 4]

# Reverse order
>>> my_list = MyList([5, 4, 3, 2, 1])
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]

# Negative numbers
>>> my_list = MyList([-2, -5, 0, 3])
>>> my_list.print_sorted()
[-5, -2, 0, 3]

# Duplicates
>>> my_list = MyList([2, 2, 1, 1, 3])
>>> my_list.print_sorted()
[1, 1, 2, 2, 3]
